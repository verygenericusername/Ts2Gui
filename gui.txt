local car
local probe

UICorner_18.Parent = ExtrasF

Frame.Active = true
Frame.Draggable = true

function findCar()
	for i, v in pairs(workspace.Vehicles:GetDescendants()) do
		if v.Name == "user" and v.Value == tostring(plr.UserId) then
			car = v.Parent
		end    
	end
	CurrentVehicle.Text = "Current Vehicle: "..car.Name
end

function findProbe()
	for i, v in pairs(workspace.Probes:GetDescendants()) do
		if v.Name == "User" and v.Value == plr.Name then
			probe = v.Parent
		end    
	end
end

function updateSpeeds()
	while wait() do
		local MPH = workspace.Storm.Tornado.Wind
		CurrentWinds.Text = "Current Winds: "..MPH.Value
	end
end

	local vehicles = workspace.Vehicles
	local hum = char:WaitForChild("Humanoid")

	ex.DestroyJointRadiusPercent = 0
	ex.BlastPressure = 5000000
	ex.BlastRadius = 10
	ex.Position = char.HumanoidRootPart.Position - Vector3.new(0,1,0)
	ex.Parent = char

	if car ~= nil then
		local char = workspace:FindFirstChild(plr.Name)
		car:MoveTo(char.HumanoidRootPart.Position)

	end
end)

local currentAnim
PlayEmote.MouseButton1Click:Connect(function()
	if EmoteID.Text ~= "" then
		local char = workspace:FindFirstChild(plr.Name)
		local hum = char.Humanoid
		local anim = Instance.new("Animation")
		anim.AnimationId = "rbxassetid://"..EmoteID.Text
		currentAnim = hum:LoadAnimation(anim)
		currentAnim:Play()
	end
end)
StopEmote.MouseButton1Click:Connect(function()
	if currentAnim ~= nil then
		currentAnim:Stop()
		currentAnim = nil
	end
end)
